openapi: 3.0.0
info:
  title: Task List API
  version: 1.0.0
  description: An API for managing tasks
paths:
  /tasks:
    get:
      summary: Get all tasks
      responses:
        '200':
          description: Successful response
          content:
            application/json:
              example:
                tasks:
                  - id: 1
                    title: Task 1
                  - id: 2
                    title: Task 2
    post:
      summary: Create a new task
      requestBody:
        required: true
        content:
          application/json:
            example:
              title: New Task
      responses:
        '201':
          description: Task created successfully
          content:
            application/json:
              example:
                id: 3
                title: New Task
  /tasks/{taskId}:
    parameters:
      - name: taskId
        in: path
        required: true
        schema:
          type: integer
    get:
      summary: Get a specific task
      responses:
        '200':
          description: Successful response
          content:
            application/json:
              example:
                id: 1
                title: Task 1
    put:
      summary: Update a task
      requestBody:
        required: true
        content:
          application/json:
            example:
              title: Updated Task 1
      responses:
        '200':
          description: Task updated successfully
          content:
            application/json:
              example:
                id: 1
                title: Updated Task 1
    delete:
      summary: Delete a task
      responses:
        '204':
          description: Task deleted successfully
